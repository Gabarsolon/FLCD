#check prime number

define int n;

read(n, stdin);

if n < 2{
    show("Not prime", stdout);
}
if n%2 == 0 AND n!=2{
    show("Not prime", stdout);
}
for d starts from 3
    transforms into d = d + 2
    stops at d*d<=n
    {
        if n%d == 0{
            show("Not prime", stdout);
        }
    }
show("Prime", stdout);
